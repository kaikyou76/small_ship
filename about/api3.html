export async function deleteFromR2(
  bucket: R2Bucket,
  fileUrl: string
): Promise<void> {
  try {
    const key = new URL(fileUrl).pathname.substring(1); // Remove leading slash
    await bucket.delete(key);
  } catch (error) {
    console.error("Failed to delete file from R2:", fileUrl, error);
    throw error;
  }
}
