## NavBar コンポーネントの作成

```tsx
//frontend/app/components/NavBar.tsx
"use client";
import { useState } from "react";
import Link from "next/link";

const NavBar = () => {
  // ユーザーのログイン状態を管理
  const [isLoggedIn, setIsLoggedIn] = useState<boolean>(false);

  // ログイン状態をトグルする関数
  const handleLoginLogout = () => {
    if (isLoggedIn) {
      // ログアウト処理
      setIsLoggedIn(false);
      // ログアウト後のリダイレクト処理（例: ログインページに移動）
    } else {
      // ログイン処理
      setIsLoggedIn(true);
      // ログイン後のリダイレクト処理（例: ホームページに移動）
    }
  };

  return (
    <nav>
      <ul>
        <li>
          <Link href="/">Home</Link>
        </li>
        <li>
          <Link href="/products">Products</Link>
        </li>
        <li>
          {isLoggedIn ? (
            <button onClick={handleLoginLogout}>Logout</button>
          ) : (
            <Link href="/login">
              <button>Login</button>
            </Link>
          )}
        </li>
      </ul>
    </nav>
  );
};

export default NavBar;
```
